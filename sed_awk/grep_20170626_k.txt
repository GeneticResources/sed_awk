#!/bin/bash
#######################################################################################################################
需求:从文件中过滤符合条件的数字(1.必须全是数字的行;2.数字位数不固定,不含空格;3.并且首位数字是大于0;4.第二,三,四位数字是数字是连续3个相同的数字;5.第五个数字跟第2,3,4位数上的数字不同.)
#######################################################################################################################
#源文件:
# echo -e '13332221  \n133333333333333333356778\n022245678\n12345567\n244456789'|cat -A
13332221  $
133333333333333333356778$
022245678$
12345567$
244456789$
#######################################################################################################################
代码:
#######################################################################################################################
echo -e '13332221  \n133333333333333333356778\n022245678\n12345567\n244456789'|grep -Po '^[1-9](\d)\1{2}(?!\1)\d+$'
244456789
echo -e '13332221  \n133333333333333333356778\n022245678\n12345567\n244456789'|grep -Po '^[1-9](?<ss>\d)\k<ss>{2}(?!\1)\d+$'
244456789
